## Word Cloud Generation

```typescript
// Topic extraction algorithm

interface Topic {
  word: string;
  count: number;
  weight: number; // For sizing in cloud
  category: string; // Most common category
}

function generateWordCloud(articles: Article[]): Topic[] {
  // 1. Extract words from titles + excerpts
  const words = articles.flatMap(a => 
    tokenize(`${a.title} ${a.contentSnippet || ''}`)
  );
  
  // 2. Remove stopwords
  const filtered = removeStopwords(words);
  
  // 3. Count frequency
  const frequency = countWords(filtered);
  
  // 4. Apply TF-IDF or similar weighting
  const weighted = applyWeighting(frequency);
  
  // 5. Extract named entities for special handling
  const entities = extractEntities(articles);
  
  // 6. Return top N topics
  return weighted
    .slice(0, 30)
    .map(w => ({
      word: w.word,
      count: w.count,
      weight: calculateFontSize(w.count, maxCount),
      category: getMostCommonCategory(w.word, articles),
    }));
}
```

## Interactions

| Action | Result |
|--------|--------|
| Tap word | Show articles containing that word |
| Tap category bar | Filter feed to that category |
| Tap entity | Show articles mentioning that entity |
| Pull down | Refresh topics |
| Change time range | Regenerate cloud |
